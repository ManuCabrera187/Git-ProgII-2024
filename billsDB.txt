CREATE PROCEDURE [dbo].[sp_GetBillById]
	@id_bill int
AS
BEGIN
	SELECT b.*, bd.precio, bd.amount, i.*
	  FROM bills b
	  JOIN bills_details bd ON bd.id_bill =b.id_bill
	  JOIN items i ON (i.id_item = bd.id_item)
	  WHERE b.id_bill = @id_bill;
END
GO

CREATE PROCEDURE [dbo].[sp_GetBills]
AS
BEGIN
	SELECT b.*,bd.precio, bd.amount, i.*
	  FROM bills b
	  JOIN bills_details bd ON bd.id_bill = b.id_bill
	  JOIN items i ON (i.id_item = bd.id_item)
	  ORDER BY b.id_bill;
END
GO

CREATE PROCEDURE [dbo].[sp_InsertBill]
	@fecha datetime,
	@client varchar(50),
	@id_bill int output
AS
BEGIN
	INSERT INTO bills(fecha, client) VALUES (GETDATE(), @client);
	SET @id_bill = SCOPE_IDENTITY();
END
